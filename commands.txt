Command for listing configuration details
    >git congif --list

Command for adding username
    >git congif --global user.name "sabarinath"

Command for adding user email
    >git congif --global user.rmail "sabarinath.py@gmail.com"

command for initializing git repository
    >git init

command for listing all untracked file
    >git status

command for adding all untracked files to staging area
    >git add -A

command for adding all staged files to local repository
    >git commit -m "message"
    
command for adding remote repository
    >git remote add name remote_path

command for adding changes in local repo to remote repo
    >git push -u remote_name branch_name
    >eg: git push -u luminar master

command for listing remote
    >git remote -v

command for branch name
    >git branch
